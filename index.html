  1. Va sur sheets.google.com → crée un nouveau tableur nommé Bacus Karaoké
  2. En-têtes ligne 1 : Horodatage | Prénom | Chanson | Artiste
  3. Menu Extensions → Apps Script, efface tout et colle ce code :
const SHEET_NAME = "Feuille 1"; function doPost(e) { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); sheet.appendRow([ new Date().toLocaleString("fr-BE"), data.name, data.song, data.artist ]); return ContentService .createTextOutput(JSON.stringify({ ok: true, row: sheet.getLastRow() })) .setMimeType(ContentService.MimeType.JSON); } function doGet() { return ContentService .createTextOutput(JSON.stringify({ status: "ok" })) .setMimeType(ContentService.MimeType.JSON); }
